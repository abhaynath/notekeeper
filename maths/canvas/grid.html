var canvas = document.getElementById("anim");
canvas.style.background = "white";
var context = canvas.getContext("2d");

function drawGrid(ctx, xMin, yMin, xMax, yMax, xStep, yStep, color, thickness) {
    const cols = Math.floor((xMax - xMin) / xStep);
    const rows = Math.floor((yMax - yMin) / yStep);
    ctx.strokeStyle = color;
    ctx.lineWidth = thickness;

    ctx.beginPath();
    for (var count = 0; count <= cols; count++) {
    ctx.moveTo(xMin + count * xStep, yMin);
    ctx.lineTo(xMin + count * xStep, yMax);
    }
    for (var count = 0; count <= rows; count++) {
    ctx.moveTo(xMin, yMin + count * yStep);
    ctx.lineTo(xMax, yMin + count * yStep);
    }

    ctx.stroke();
}
drawGrid(context, 10, 10, canvas.width - 10, canvas.height - 10, 10, 10, "#5288ff", 1);

Reactive Block


<script>
export let name = "";
$: greeting = "Hello, " + name + "!";
let radius = 5,
    diameter = 0,
    circumference = 0,
    area = 0;
$: {
    diameter = 2 * radius;
    circumference = Math.PI * diameter;
    area = Math.PI * radius * radius;
}
</script>

<div class="test">
<input type="text" bind:value={name} />
<input type="number" bind:value={radius} />
<div>
    <div>Radius : {radius}</div>
    <div>Diameter : {diameter.toFixed(2)}</div>
    <div>Circumference : {circumference.toFixed(2)}</div>
    <div>Area : {area.toFixed(2)}</div>
</div>
<div>{greeting}</div>
</div>

<style>
.test {
    padding: 10px;
    border: 1px solid gray;
}
</style>

Slide Transition


<script>
import { slide } from "svelte/transition";
let arr = ["Rohit", "Shikhar", "Virat", "Shreyas", "Hardik", "Jaspreet"];
let count = 0;
let isVisible = true;
</script>

<div>
  <input type="range" bind:value={count} max={arr.length} />{count}
  <label><input type="checkbox" bind:checked={isVisible} /> Show Players</label>
  {#if isVisible}
    {#each arr.slice(0, count) as player}
      <div class="player" transition:slide>{player}</div>
    {/each}
  {/if}
</div>

<style>
  .player {
    border: 1px solid #ccc;
    text-align: center;
    padding: 5px;
  }
</style>

Tweening


<script>
  import { cubicOut } from "svelte/easing";
  import { tweened } from "svelte/motion";
  const t = tweened(0, {
    duration: 400,
    easing: cubicOut,
  });
</script>

<progress value={$t} />
<div class="btn">
  <button on:click={() => t.set(0)}>0</button>
  <button on:click={() => t.set(0.25)}>25</button>
  <button on:click={() => t.set(0.5)}>50</button>
  <button on:click={() => t.set(0.75)}>75</button>
  <button on:click={() => t.set(1)}>100</button>
</div>

<style>
  progress {
    width: 100%;
  }
  .btn {
    display: flex;
    justify-content: center;
  }
</style>

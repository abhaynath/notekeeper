Batting.svelte


<script>
import { teamScore } from "../stores/store.js";

let arr = [
{ id: "id1", name: "Dot", run: 0 },
{ id: "id2", name: "1", run: 1 },
{ id: "id3", name: "2", run: 2 },
{ id: "id4", name: "3", run: 3 },
{ id: "id5", name: "4", run: 4 },
{ id: "id6", name: "6", run: 6 },
];

const ballClick = (val) => {
console.log(teamScore);
teamScore.update((obj) => {
    obj.runs = obj.runs + val.run;
    obj.balls = obj.balls + 1;
    return obj;
});
};
</script>

<div class="batting">
{#each arr as ball (ball.id)}
<div
    class="ball"
    on:click={() => {
    ballClick(ball);
    }}
>
    {ball.run}
</div>
{/each}
</div>

Bowling.svelte


<script>
import { teamScore } from "../stores/store.js";

const addWicket = () => {
teamScore.update((obj) => {
    obj.wicket++;
    obj.balls++;
    return obj;
});
};
</script>

<div class="wicket" on:click={addWicket}>Wicket!</div>

<style>
  .wicket {
    background: green;
    color: yellow;
    padding: 10px;
    display: inline-block;
    cursor: pointer;
  }
</style>

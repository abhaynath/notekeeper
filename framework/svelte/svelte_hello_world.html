Svelte

Simple hello world application in Svelte

  1. Run following command to create Svelte App 
        npx degit sveltejs/template my-app
    cd my-app
  2. To use TypeScript run following command. 
    node scripts/setupTypeScript.js
  3. Install all packages 
    npm install --verbose
  4. To run the development server, 
    npm run dev
    To change the default port, change 'package.json' scripts.dev settings. 
     "dev": "SET PORT=5474 && rollup -c -w",
  5. To build the project 
    npm run build

Using Props

App.svelte


<script>
import TestFile from "./components/TestFile.svelte";

let players = [];
function playerAdded(val) {
val = val.trim();
if (val.length < 1) {
    return;
}
players.push(val);
players = players;
}
</script>

<div class="main">
  <TestFile title="My Test App" {playerAdded} />
  <h3>Parent</h3>
  <ol>
    {#each players as player (player)}
      <li>{player}</li>
    {/each}
  </ol>
</div>

<style>
  .main {
    padding: 10px;
  }
</style>

TestFile.svelte


<script>
export let title = "";

export let playerAdded;

let playerName = "";

const clickHandler = () => {
playerAdded(playerName);
playerName = "";
};
</script>

<div>
  <div class="tl">{title}</div>
  <input type="text" bind:value={playerName} />
  <button
    on:click={() => {
      clickHandler();
    }}>Send to Parent</button
  >
</div>

<style>
  .tl {
    font-weight: bold;
    font-size: 18px;
  }
</style>

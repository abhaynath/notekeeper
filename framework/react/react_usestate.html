1. update state values on the basis of previous state value

when we have to update the state based on the previous value of state, instead of direct assignment of state, use function 

    import React, { useState } from "react";

    const CompUseState = () => {
        const [count, setCount] = useState(0);
        return (
        <div>
            <h3>Count = {count}</h3>
            <div>
            <button
                onClick={() => {
                setCount((prev) => prev + 1);
                }}
            >
                Increment by 1
            </button>
            </div>
            <div>
            <button
                onClick={() => {
                setCount((prev) => prev - 1);
                }}
            >
                Decrement by 1
            </button>
            </div>
        </div>
        );
    };
    
    export default CompUseState;

2. useState with array of objects

Following code creates a random object on button click and push it to state array. On click of the object box, it removes from the state array. 
    import React, { useState } from "react";

    const CompUseStateWithArray = () => {
        const [arr, setArray] = useState<any>([]);
        return (
        <div className="box">
            <h3>useState using Array state</h3>
            <p>click on the number to remove it from array.</p>
            <h3>Array of random object</h3>
            <div className="list">
            {arr.map((item: any) => {
                return (
                <div
                    key={item}
                    onClick={() => {
                    let tempArr = [...arr];
                    let index = tempArr.findIndex((k: any) => k.id == item.id);
                    if (index != -1) {
                        tempArr.splice(index, 1);
                        setArray(tempArr);
                    }
                    }}
                >
                    {item.name}
                </div>
                );
            })}
            </div>
            <div>
            <button
                onClick={() => {
                let num = Math.floor(Math.random() * 99999999);
                let obj = { id: num, name: num };
                setArray((prevState: any) => [...prevState, obj]);
                }}
            >
                Add random element
            </button>
            </div>
        </div>
        );
    };
    
    export default CompUseStateWithArray;

3. useState with object

    import React, { useState } from "react";

    const CompUseStateWithObject = () => {
    const [userDetails, setUserDetails] = useState({
        firstName: "",
        lastName: "",
    });
    return (
        <div className="box">
        <h3>useState using object state</h3>
        <p>
            When we use object as a state, we have to destructure the object
            properties
        </p>
        <h3>User Details</h3>
        <div>
            <b>{userDetails.firstName + " " + userDetails.lastName}</b>
        </div>
        <div>
            <input
            type="text"
            placeholder="First Name"
            onChange={(e) => {
                setUserDetails({ ...userDetails, firstName: e.target.value });
            }}
            ></input>
            <input
            type="text"
            placeholder="Last Name"
            onChange={(e) => {
                setUserDetails({ ...userDetails, lastName: e.target.value });
            }}
            ></input>
        </div>
        </div>
    );
    };

    export default CompUseStateWithObject;

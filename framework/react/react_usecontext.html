1. UseContext example

When we have to pass data to deeply nested component, we can use useContext. 

  import React from "react";
  import CompC from "./compC";

  export const UserContext = React.createContext<any>(null);
  const CompUseContext = () => {
  const userData = { name: "Sachin Tendulkar", city: "Mumbai" };
  return (
      <div>
      <UserContext.Provider value={userData}>
          <CompC></CompC>
      </UserContext.Provider>
      </div>
  );
  };

  export default CompUseContext;
        

  import React, { useContext } from "react";
  import { UserContext } from ".";
  
  const CompC = () => {
      const userData = useContext(UserContext);
      console.log("--userData--", userData);
      return (
      <div>
          <h3>Comp C</h3>
          <div>{userData.name} </div>
          <div>{userData.city} </div>
      </div>
      );
  };
  
  export default CompC;

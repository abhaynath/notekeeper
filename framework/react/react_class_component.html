Data Attributes

Using jQuery

    <ul>
        <li data-snake="anaconda">Anaconda</li>
        <li data-snake="cobra">King Kobra</li>
        <li data-snake="mamba">Black Mamba</li>
    </ul>
                
    $("li").click(function (e) {
        e.stopPropagation();
        
        // Reading data Attributes
        var snakeType = $(this).attr("data-snake");

        //Setting value of data attribute
        $(this).attr('data-snake', "rattle snake");

        console.log(snakeType);

    });

Using JavaScript


    var links = document.getElementsByTagName("li");
    for (var count = 0; count < links.length; count++) {
        links[count].onclick = show;
    }

    function show(e) {
        e.stopImmediatePropagation();

        // Reading data Attributes
        var snakeType = e.target.dataset.snake;

        //Setting value of data attribute
        e.target.dataset.snake = "rattle snake";

        console.log(snakeType);

    }

Checkbox

To create checkbox, 

    <input type="checkbox" id="chkCountry" checked><label for="chkCountry">Indian</label>
    <div class="output" id="chk_output"></div>
                

    $("#chkCountry").change(function (e) {
        e.stopPropagation();
        var chk_status = $(this).prop("checked");
        if (chk_status) {
            $("#chk_output").html("Checked !");
        } else {
            $("#chk_output").html("Unchecked");
        }


    });

Get list of all checked checkboxes


    <ol id="skills">
        <li><input type="checkbox" id="row_0" data-skill="js"><label for="row_0">JavaScript</label></li>
        <li><input type="checkbox" id="row_1" data-skill="cs"><label for="row_1">C#</label></li>
        <li><input type="checkbox" id="row_2" data-skill="py"><label for="row_2">Python</label></li>
        <li><input type="checkbox" id="row_3" data-skill="sql"><label for="row_3">SQL</label></li>
    </ol>
    <button id="submit">Submit</button>

                

    $("#submit").click(function (e) {
        e.stopPropagation();

        var arrSkills = [];

        $.each($("#skills input[type='checkbox']:checked"), function () {
            var skill = $(this).attr("data-skill");
            arrSkills.push(skill);

        });

        console.log(arrSkills);
    });

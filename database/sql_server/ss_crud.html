  create table player
  (
      	id bigint identity(1,1) not null primary key,
      	firstName varchar(50),
      	middleName varchar(50),
      	lastName varchar(50),
      	teamId bigint,
      	birthDate date,
      	birthPlace varchar(100),
      	height int,
      	roleId bigint,
      	battingStyle bigint,
      	bowlingStyle bigint,
      	info text

  )

Create Tables with Foreign Key constraints


    create table tournament
    (
        id bigint identity(1,1) not null primary key,
        title varchar(150) unique,
        editionOf bigint,
        info text,
        startDate date,
        endDate date,
        manOfSeries bigint,
        constraint fk_man_of_series foreign key (manOfSeries)
        references player(id)
        on delete cascade
        on update cascade
    
    )

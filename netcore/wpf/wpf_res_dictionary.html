Resource Dictionary

Right-click on a project and add a resource dictionary and ResourceDictionaryMargin.xaml

<ResourceDictionary xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
        xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml" xmlns:clr="clr-namespace:System;assembly=netstandard">
  <clr:String x:Key="authorName">Abhay</clr:String>
  <SolidColorBrush x:Key="AppBG" Color="Aqua"></SolidColorBrush>
</ResourceDictionary>

Once you are finished with defining a ResourceDictionary, you can add it inside a App.xaml if you wanted to use it across the application.

App.xaml

<Application x:Class="WPF_Scratchpad.App"
  xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
  xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
  xmlns:local="clr-namespace:WPF_Scratchpad"
  xmlns:clr="clr-namespace:System;assembly=mscorlib"
  StartupUri="MainWindow.xaml">
<Application.Resources>
  <ResourceDictionary>
  <ResourceDictionary.MergedDictionaries>
      <ResourceDictionary Source="ResourceDictionaryMargin.xaml"></ResourceDictionary>
  </ResourceDictionary.MergedDictionaries>
  </ResourceDictionary>
</Application.Resources>
</Application>

To use this values in XAML,

 <StackPanel>
  <TextBlock Text="{StaticResource authorName}"></TextBlock>
  <TextBlock Name="TxtAuthorName"></TextBlock>
</StackPanel>

To access properties via code,

using System.Windows;
using System.Windows.Controls;

namespace WPF_Scratchpad.UserControls
{
    public partial class LoginControl : UserControl
    {
        public LoginControl()
        {
            InitializeComponent();
            TxtAuthorName.Text = Application.Current.FindResource("AppBG").ToString();
        }

    }
}

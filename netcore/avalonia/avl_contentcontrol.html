ControlSwitchPage.axaml


<Window xmlns="https://github.com/avaloniaui"
        xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
        xmlns:d="http://schemas.microsoft.com/expression/blend/2008"
        xmlns:mc="http://schemas.openxmlformats.org/markup-compatibility/2006"
        xmlns:vm="using:UI.ViewModels"
        xmlns:views="using:UI.Views.Controls"
        mc:Ignorable="d" d:DesignWidth="800" d:DesignHeight="450"
        xmlns:i="https://github.com/projektanker/icons.avalonia"
        x:Class="UI.Views.Pages.ControlSwitchPage"
        x:DataType="vm:ControlSwitchViewModel"
        WindowState="Maximized"
        Icon="/Assets/Icons/home72.ico"
        Title="Content Switch Demo">

  <Design.DataContext>
    <vm:ControlSwitchViewModel></vm:ControlSwitchViewModel>
  </Design.DataContext>

  <Window.DataTemplates>
    <DataTemplate DataType="{x:Type vm:HomeViewModel}">
      <views:HomeControl></views:HomeControl>
    </DataTemplate>
    <DataTemplate DataType="{x:Type vm:PorfolioViewModel}">
      <views:PortfolioControl></views:PortfolioControl>
    </DataTemplate>
    <DataTemplate DataType="{x:Type vm:TradingViewModel}">
      <views:TradingControl></views:TradingControl>
    </DataTemplate>
  </Window.DataTemplates>

  <Window.Styles>
    <Style Selector="StackPanel Button">
      <Setter Property="Background" Value="Navy"></Setter>
      <Setter Property="Foreground" Value="White"></Setter>
      <Setter Property="HorizontalAlignment" Value="Center"></Setter>
      <Setter Property="BorderBrush" Value="White"></Setter>
      <Setter Property="Margin" Value="4"></Setter>
    </Style>
  </Window.Styles>

  <Grid Margin="10" Background="SeaGreen">

    <Grid.ColumnDefinitions>
      <ColumnDefinition Width="Auto"></ColumnDefinition>
      <ColumnDefinition Width="2"></ColumnDefinition>
      <ColumnDefinition Width="*"></ColumnDefinition>
    </Grid.ColumnDefinitions>

    <Grid.RowDefinitions>
      <RowDefinition></RowDefinition>
    </Grid.RowDefinitions>

    <StackPanel Grid.Column="0" Grid.Row="0" Orientation="Vertical"  Spacing="10" HorizontalAlignment="Center" VerticalAlignment="Center">
      <Button Command="{Binding GotoHomeViewCommand}">
        <i:Icon Value="fa fa-home"></i:Icon>
      </Button>
      <Button Command="{Binding GoToPortfolioViewCommand}">Portfolio</Button>
      <Button Command="{Binding GoToTradingViewCommand}">Trading</Button>
    </StackPanel>

    <GridSplitter Grid.Column="1" Grid.Row="0" Background="Orange" ></GridSplitter>

    <ContentControl Grid.Column="2" Grid.Row="0" Background="AliceBlue" Content="{Binding SelectedView}" ></ContentControl>

  </Grid>
</Window>

using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;

namespace UI.ViewModels;

public partial class ControlSwitchViewModel:ViewModelBase
{
    [ObservableProperty]
    private ViewModelBase _selectedView;
    public ControlSwitchViewModel()
    {
        SelectedView = new HomeViewModel();
    }

    [RelayCommand]
    private void GoToPortfolioView()
    {
        SelectedView = new PorfolioViewModel();
    }
    [RelayCommand]
    private void GoToTradingView()
    {
        SelectedView = new TradingViewModel();
    }
    [RelayCommand]
    private void GotoHomeView()
    {
        SelectedView = new HomeViewModel();
    }
}

    package slicedemo

    import (
    	"fmt"
    	"reflect"
    )
    
    func SliceDemo() {
    	fmt.Println("----Slice demo----")
    
    	var intSlice []int
    
    	fmt.Println(reflect.ValueOf(intSlice).Kind())
    	intSlice = append(intSlice, 10)
    
    	fmt.Printf("%v\n", intSlice)
    
    	//-----------------------------------
    
    	var intSlice1 = []int{3, 45, 66}
    	var strSlice1 = []string{"Rohit", "Shikhar", "Virat", "Dhoni", "Ajinkya", "Bhuvi"}
    	fmt.Println(len(strSlice1))
    	fmt.Println(cap(strSlice1))
    	fmt.Printf("%v \n", intSlice1)
    
    	//-----------------------------------
    
    	var intSlice2 = new([50]int)[0:10]
    	fmt.Printf("intSlice2 \t len :%v \t cap %v \n", len(intSlice2), cap(intSlice2))
    
    	//-----------------------------------
    
    	var intSlice3 = make([]int, 10)    //when length and capacity is same
    	var intSlice4 = make([]int, 5, 10) //when length and capacity is different
    
    	fmt.Println(len(intSlice3), " -- ", len(intSlice4))
    
    	// strSlice2 := removeIndex(strSlice1, 0)
    	// fmt.Println(strSlice2)
    
    	fmt.Println(strSlice1)
    	//--------------loop 1---------------------
    	fmt.Println("--------------loop 1---------------------")
    	for index, element := range strSlice1 {
    		fmt.Println(index, "  ", element)
    	}
    
    	//--------------loop 2---------------------
    	fmt.Println("--------------loop 2---------------------")
    	for _, element := range strSlice1 {
    		fmt.Println(element)
    	}
    	fmt.Println("--------------loop 3---------------------")
    
    	j := 0
    	for range strSlice1 {
    		fmt.Println(strSlice1[j])
    		j++
    	}
    	strSlice1 = removeIndex(strSlice1, 0)
    	fmt.Println("---after removing---")
    	fmt.Println(strSlice1)
    }
    func removeIndex(s []string, index int) []string {
    	return append(s[:index], s[index+1:]...)
    
    }

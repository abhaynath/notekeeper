Go Testing

  1. Write a function which need to be tested. Function name begins with Capital letter so that it is exported and can be used in test file.
  2. Create a new test file. Test file name must end with _test.go
  3. In test file, add a new function with name must be Test(Function name)
  4. Run following command to run test suite 
    go test -v
  5. calculator.go 
    
    package main

    func Divide(x int, y int) int {
    	if y == 0 {
    		return 0
    	}
    	return x / y
    }
  6. calculator_test.go 
    
    package main

    import "testing"

    func TestDivide(t *testing.T) {
    	type args struct {
    		x int
    		y int
    	}
    	tests := []struct {
    		name string
    		args args
    		want int
    	}{
    		{"divides integer properly", args{60, 5}, 512},
    		{"divides integer properly", args{60, 0}, 0},
    	}
    	for _, tt := range tests {
    		t.Run(tt.name, func(t *testing.T) {
    			if got := Divide(tt.args.x, tt.args.y); got != tt.want {
    				t.Errorf("Divide() = %v, want %v", got, tt.want)
    			}
    		})
    	}
    }

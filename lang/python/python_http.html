Simple Get request


import requests

url = "https://jsonplaceholder.typicode.com/posts/1"

api_response = requests.get(url)

print(f"Status = {api_response.status_code}")       # Status = 200
text_content = api_response.text

print(type(text_content))                           # <class 'str'>
print(text_content)

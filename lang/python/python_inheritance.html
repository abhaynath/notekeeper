Inheritance

Python has a super() function that will make the child class inherit all the methods and properties from its parent


class Person:
    def __init__(self, name, age) -> None:
        self.name = name
        self.age = age

    def show(self):
        print(f"Name = {self.name} Age = {self.age}")


class Student(Person):
    def __init__(self, name, age) -> None:
        super().__init__(name, age)
        self.graduationYear = 2005

    def showStudent(self):
        print(
            f"Name = {self.name} Age = {self.age} Graduation Year = {self.graduationYear}")


abhay = Student("Abhay", 35)
abhay.show()                        # Name = Abhay Age = 35
abhay.showStudent()                 # Name = Abhay Age = 35 Graduation Year = 2005

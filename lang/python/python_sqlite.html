class Employee:

    def __init__(self, id, first, last, pay) -> None:
        self.id = id
        self.first = first
        self.last = last
        self.pay = pay

    @property
    def email(self) -> str:
        return f"{self.first}.{self.last}@email.com"

    @property
    def fullname(self) -> str:
        return f"{self.first} {self.last}"

    def __repr__(self) -> str:
        return f"Employee : {self.id} {self.fullname} {self.email} {self.pay}"

import sqlite3
from employee import Employee


EMPLOYEE_TABLE = {"DB_NAME": "employee.db", "TABLE_NAME": "employee", "ID": "id",
                  "FIRST_NAME": "first", "LAST_NAME": "last", "PAY": "pay"}

conn = sqlite3.connect(EMPLOYEE_TABLE["DB_NAME"])

c = conn.cursor()

emp1 = Employee(1, 'sachin', 'tendulkar', 10000)
emp2 = Employee(2, 'rohit', 'sharma', 50000)

# create table
c.execute(
    f"CREATE TABLE IF NOT EXISTS {EMPLOYEE_TABLE['TABLE_NAME']} ({EMPLOYEE_TABLE['ID']} integer primary key, {EMPLOYEE_TABLE['FIRST_NAME']} text, {EMPLOYEE_TABLE['LAST_NAME']} text,  {EMPLOYEE_TABLE['PAY']} integer)")


# insert employee 1 data
c.execute(f"INSERT INTO {EMPLOYEE_TABLE['TABLE_NAME']}({EMPLOYEE_TABLE['FIRST_NAME']},{EMPLOYEE_TABLE['LAST_NAME']},{EMPLOYEE_TABLE['PAY']}) VALUES (?,?,?)",
          (emp1.first, emp1.last, emp1.pay))

# insert employee 2 data
c.execute(f"INSERT INTO {EMPLOYEE_TABLE['TABLE_NAME']}({EMPLOYEE_TABLE['FIRST_NAME']},{EMPLOYEE_TABLE['LAST_NAME']},{EMPLOYEE_TABLE['PAY']}) VALUES (?,?,?)",
          (emp2.first, emp2.last, emp2.pay))


# get all employee data
c.execute(
    f"SELECT {EMPLOYEE_TABLE['ID']}, {EMPLOYEE_TABLE['FIRST_NAME']}, {EMPLOYEE_TABLE['PAY']} FROM {EMPLOYEE_TABLE['TABLE_NAME']}")

rows = c.fetchall()

for r in rows:
    print(r[0], r[1], r[2])


conn.commit()

conn.close()

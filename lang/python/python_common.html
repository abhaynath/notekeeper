Generate Unique id string (UUID)

uuid.uuid4() method

To create a random id, you call the uuid4 () method and it will automatically generate a unique id.


import uuid

id2 = str(uuid.uuid4())

print(id2)

Creating unique Id in constructor


import uuid


class Player:
    def __init__(self, name) -> None:
        self.id = str(uuid.uuid4())
        self.first_name = name

    def display(self):
        print(f"{self.id} {self.first_name.title()}")


names = ["sachin", "saurav", "rahul"]

team = list()

for name in names:
    player = Player(name)
    team.append(player)


for p in team:
    p.display()

Swap pointers

#include <stdio.h>
void swapPointers(int *x, int *y)
{
    int t;
    t = *x;
    *x = *y;
    *y = t;
}
int main()
{
    int a = 10, b = 20;
    printf("a = %d and b = %d\n", a, b);
    swapPointers(&a, &b);
    printf("a = %d and b = %d\n", a, b);
    return 0;
}

Call by reference example

#include <stdio.h>
void calculate(int, float *, float *);

int main()
{
    int radius;
    float area, perimeter;

    printf("Please enter radius\n");
    scanf("%d", &radius);

    calculate(radius, &area, &perimeter);

    printf("Radius = %d\n", radius);
    printf("Area = %f\n", area);
    printf("Circumference = %f\n", perimeter);

    return 0;
}
void calculate(int r, float *a, float *p)
{
    *a = 3.1416 * r * r;
    *p = 2 * 3.1616 * r;
}

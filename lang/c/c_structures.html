Structure Basics

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct book
{
    char *name;
    float price;
    int pages;
};

// Function to print book details
void printBook(const struct book *b)
{
    printf("Book Details : %s %.2f %d\n", b->name, b->price, b->pages);
}

int main()
{
    struct book b1 = {"Let us C", 300.50, 450};
    printBook(&b1);

    struct book b2;
    char temp[100];

    printf("Enter book name : ");
    fgets(temp, sizeof(temp), stdin);
    temp[strcspn(temp, "\n")] = '\0'; // Remove newline

    b2.name = malloc(strlen(temp) + 1);
    if (!b2.name)
    {
        printf("Memory allocation failed!\n");
        return 1;
    }
    strcpy(b2.name, temp);

    printf("Enter book price : ");
    scanf("%f", &b2.price);

    printf("Enter book pages : ");
    scanf("%d", &b2.pages);

    printBook(&b2);

    free(b2.name);
    return 0;
}

Nested Structures

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct address
{
    char *city;
    char *pincode;
};
struct employee
{
    char *name;
    int empCode;
    struct address location;
};

void display(const struct employee *);

int main()
{
    struct employee e1 = {"Abhay Nath", 35, "Mumbai", "400067"};
    display(&e1);
    return 0;
}
void display(const struct employee *e)
{
    printf("Name %s\n", e->name);
    printf("Employee Code %d\n", e->empCode);
    printf("City %s\n", e->location.city);
    printf("Pin %s\n", e->location.pincode);
}

Vector Operations using structures

#include <stdio.h>
#include <math.h>

typedef struct
{
    double x;
    double y;
    double z; // Supports 3D vectors, set z = 0 for 2D
} Vector;

// Function to display a vector
void printVector(Vector v)
{
    printf("(%.3f, %.3f, %.3f)\n", v.x, v.y, v.z);
}

// Vector addition
Vector add(Vector a, Vector b)
{
    Vector result = {a.x + b.x, a.y + b.y, a.z + b.z};
    return result;
}

// Vector subtraction
Vector subtract(Vector a, Vector b)
{
    Vector result = {a.x - b.x, a.y - b.y, a.z - b.z};
    return result;
}

// Scalar multiplication
Vector scalarMultiply(Vector v, double scalar)
{
    Vector result = {v.x * scalar, v.y * scalar, v.z * scalar};
    return result;
}

// Dot product
double dotProduct(Vector a, Vector b)
{
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

// Cross product (only valid for 3D)
Vector crossProduct(Vector a, Vector b)
{
    Vector result = {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x};
    return result;
}

// Magnitude
double magnitude(Vector v)
{
    return sqrt(v.x * v.x + v.y * v.y + v.z * v.z);
}

// Normalization
Vector normalize(Vector v)
{
    double mag = magnitude(v);
    if (mag == 0)
        return v; // Avoid division by zero
    return scalarMultiply(v, 1.0 / mag);
}

int main()
{
    Vector v1 = {3, 4, 0}; // 2D vector (z=0)
    Vector v2 = {1, 2, 0};

    printf("Vector 1: ");
    printVector(v1);
    printf("Vector 2: ");
    printVector(v2);

    printf("\nAddition: ");
    printVector(add(v1, v2));

    printf("Subtraction: ");
    printVector(subtract(v1, v2));

    printf("Scalar Multiplication (v1 × 2): ");
    printVector(scalarMultiply(v1, 2));

    printf("Dot Product: %.3f\n", dotProduct(v1, v2));

    printf("Magnitude of v1: %.3f\n", magnitude(v1));

    printf("Normalized v1: ");
    printVector(normalize(v1));

    // Example for cross product (3D)
    Vector v3 = {1, 0, 0};
    Vector v4 = {0, 1, 0};
    printf("\nCross Product of v3 and v4: ");
    printVector(crossProduct(v3, v4));

    return 0;
}

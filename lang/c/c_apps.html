#include <stdio.h>
#include <wchar.h>
#include <locale.h>

void print_binary(unsigned int codepoint)
{
    int total_bits = (codepoint <= 0xFF) ? 8 : (codepoint <= 0xFFFF) ? 16
                                                                     : 32;

    for (int i = total_bits - 1; i >= 0; i--)
    {
        printf("%d", (codepoint >> i) & 1);
        if (i % 4 == 0 && i != 0) // space after every 4 bits except last
            printf(" ");
    }
}

int main()
{
    setlocale(LC_ALL, "");

    wchar_t ch;

    wprintf(L"Enter a character: ");
    if (wscanf(L"%lc", &ch) != 1)
    {
        wprintf(L"Invalid input.\n");
        return 1;
    }

    unsigned int codepoint = (unsigned int)ch;

    wprintf(L"Character: %lc\n", ch);
    printf("Unicode codepoint: U+%04X\n", codepoint);

    if (codepoint <= 0x7F)
        printf("ASCII value: %u\n", codepoint);
    else
        printf("ASCII value: (Not an ASCII character)\n");

    printf("Binary: ");
    print_binary(codepoint);
    printf("\n");

    printf("Hexadecimal: 0x%X\n", codepoint);
    printf("Octal: 0%o\n", codepoint);

    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main()
{
    char buffer[1024];
    FILE *fp;
    int git_found = 0;

#ifdef _WIN32
    const char *command = "cmd /c git --version 2>&1";
#else
    const char *command = "git --version 2>&1";
#endif

    fp = popen(command, "r");
    if (fp == NULL)
    {
        perror("popen failed");
        return 1;
    }

    printf("Checking Git installation...\n\n");

    while (fgets(buffer, sizeof(buffer), fp) != NULL)
    {
        printf("%s", buffer);
        if (strstr(buffer, "git version") != NULL)
        {
            git_found = 1;
        }
    }

    int status = pclose(fp);

    if (!git_found || status != 0)
    {
        printf("\n❌ Git is not installed or not found in PATH.\n");
        return 1;
    }

    printf("\n✅ Git is installed.\n");
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#if defined(_WIN32) || defined(_WIN64)
#include <windows.h>
#include <intrin.h>
#elif __linux__ || __APPLE__
#include <sys/utsname.h>
#include <unistd.h>
#include <sys/statvfs.h>
#endif

#if __linux__
#include <sys/sysinfo.h>
#elif __APPLE__
#include <sys/sysctl.h>
#endif

void print_user_info()
{
    char username[256] = {0};
    char hostname[256] = {0};
    char *home = NULL;

#if defined(_WIN32) || defined(_WIN64)
    DWORD size = sizeof(username);
    GetUserNameA(username, &size);

    DWORD hostSize = sizeof(hostname);
    GetComputerNameA(hostname, &hostSize);

    home = getenv("USERPROFILE");
#else
    getlogin_r(username, sizeof(username));
    gethostname(hostname, sizeof(hostname));
    home = getenv("HOME");
#endif

    printf("=== User Information ===\n");
    printf("Username: %s\n", username);
    printf("Hostname: %s\n", hostname);
    if (home)
        printf("Home Dir: %s\n", home);
}

void print_gpu_info()
{
#if defined(_WIN32) || defined(_WIN64)
    printf("\n=== GPU Information ===\n");
    system("wmic path win32_VideoController get Name");
#elif __linux__
    printf("\n=== GPU Information ===\n");
    if (system("command -v lspci >/dev/null 2>&1") == 0)
        system("lspci | grep -i 'vga\\|3d\\|display'");
    else
        printf("lspci not available\n");
#elif __APPLE__
    printf("\n=== GPU Information ===\n");
    system("system_profiler SPDisplaysDataType | grep 'Chipset Model'");
#endif
}

void print_disk_info()
{
#if defined(_WIN32) || defined(_WIN64)
    ULARGE_INTEGER freeBytesAvailable, totalBytes, freeBytes;
    if (GetDiskFreeSpaceEx("C:\\", &freeBytesAvailable, &totalBytes, &freeBytes))
    {
        printf("\n=== Disk Information ===\n");
        printf("Total Disk Size: %llu GB\n", totalBytes.QuadPart / (1024ULL * 1024 * 1024));
        printf("Free Disk Space: %llu GB\n", freeBytes.QuadPart / (1024ULL * 1024 * 1024));
    }
#elif __linux__ || __APPLE__
    struct statvfs stat;
    if (statvfs("/", &stat) == 0)
    {
        unsigned long long total = (unsigned long long)stat.f_blocks * stat.f_frsize;
        unsigned long long free = (unsigned long long)stat.f_bfree * stat.f_frsize;
        printf("\n=== Disk Information ===\n");
        printf("Total Disk Size: %llu GB\n", total / (1024ULL * 1024 * 1024));
        printf("Free Disk Space: %llu GB\n", free / (1024ULL * 1024 * 1024));
    }
#endif
}

void print_system_info()
{
#if defined(_WIN32) || defined(_WIN64)
    SYSTEM_INFO si;
    MEMORYSTATUSEX statex;
    GetSystemInfo(&si);
    statex.dwLength = sizeof(statex);
    GlobalMemoryStatusEx(&statex);

    printf("\n=== System Information (Windows) ===\n");
    printf("Processor Architecture: %u\n", si.wProcessorArchitecture);
    printf("Number of Processors: %u\n", si.dwNumberOfProcessors);
    printf("Total Physical Memory: %llu MB\n", statex.ullTotalPhys / (1024 * 1024));

    int cpuInfo[4] = {0};
    char cpuBrand[0x40] = {0};
    __cpuid(cpuInfo, 0x80000002);
    memcpy(cpuBrand, cpuInfo, sizeof(cpuInfo));
    __cpuid(cpuInfo, 0x80000003);
    memcpy(cpuBrand + 16, cpuInfo, sizeof(cpuInfo));
    __cpuid(cpuInfo, 0x80000004);
    memcpy(cpuBrand + 32, cpuInfo, sizeof(cpuInfo));
    printf("CPU Model: %s\n", cpuBrand);

#elif __linux__
    struct utsname unameData;
    uname(&unameData);
    struct sysinfo info;
    sysinfo(&info);

    printf("\n=== System Information (Linux) ===\n");
    printf("System: %s %s\n", unameData.sysname, unameData.release);
    printf("Machine: %s\n", unameData.machine);
    printf("Total RAM: %ld MB\n", info.totalram / (1024 * 1024));

    FILE *cpuinfo = fopen("/proc/cpuinfo", "r");
    if (cpuinfo)
    {
        char line[256];
        while (fgets(line, sizeof(line), cpuinfo))
        {
            if (strncmp(line, "model name", 10) == 0)
            {
                printf("CPU Model: %s", strchr(line, ':') + 2);
                break;
            }
        }
        fclose(cpuinfo);
    }

#elif __APPLE__
    struct utsname unameData;
    uname(&unameData);

    printf("\n=== System Information (macOS) ===\n");
    printf("System: %s %s\n", unameData.sysname, unameData.release);
    printf("Machine: %s\n", unameData.machine);

    int64_t memsize;
    size_t len = sizeof(memsize);
    sysctlbyname("hw.memsize", &memsize, &len, NULL, 0);
    printf("Total RAM: %lld MB\n", memsize / (1024 * 1024));

    char cpuBrand[256];
    len = sizeof(cpuBrand);
    sysctlbyname("machdep.cpu.brand_string", &cpuBrand, &len, NULL, 0);
    printf("CPU Model: %s\n", cpuBrand);
#endif
}

int main()
{
    print_user_info();
    print_system_info();
    print_gpu_info();
    print_disk_info();
    return 0;
}

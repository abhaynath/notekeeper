G++ setup

Setting up compiler
  1. Navigate to MSYS2 website
  2. Download the installer for your environment. It might look like this release for windows
  3. After setup is complete, it will show MSYS2 terminal. Run following command 
    pacman -S mingw-w64-x86_64-make
  4. Add installation path in windows environment variables. C:\msys64\mingw64\bin
  5. Check the installation 
    gcc --version
g++ --version
gdb --version

project(CmakeCpp VERSION 1.0 LANGUAGES CXX)

# Add library from lib/
add_subdirectory(lib)

# Add executable from src/
add_executable(CmakeCpp src/main.cpp)

# Link the library
target_link_libraries(CmakeCpp PRIVATE MathUtils)

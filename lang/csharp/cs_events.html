Custom Event Example


namespace ConsoleRnD
{
    public class MyCustomButton
    {
        public event EventHandler<ButtonPressedEventArgs>? ButtonPressed;

        public void OnButtonPressed(char keyCode)
        {
            ButtonPressed?.Invoke(this, new ButtonPressedEventArgs(keyCode));
        }
    }
    public class ButtonPressedEventArgs
    {
        public ButtonPressedEventArgs(char keyCode)
        {
            KeyCode = keyCode;
        }
        public char KeyCode { get; }
    }
    public class Program
    {
        static void Main(string[] args)
        {
            var btnLogin = new MyCustomButton();
            btnLogin.ButtonPressed += LoginClicked;

            var keyCode = Console.ReadKey(true).KeyChar;
            btnLogin.OnButtonPressed(keyCode);

        }

        private static void LoginClicked(object? sender, ButtonPressedEventArgs e)
        {
            Console.WriteLine($"Button {e.KeyCode} pressed ");
        }
    }
}

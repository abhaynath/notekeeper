        async function postData(url = "", data = {}) {
            const response = await fetch(url, {
            method: "POST",
            mode: "cors",
            headers: {
                "Content-Type": "application/json",
                "Access-Control-Allow-Origin": "*",
                Authorization: "Bearer " + token,
            },
    
            body: JSON.stringify(data),
            });
            return response.json();
        }

    <!DOCTYPE html>
    <html lang="en">
        <head>
        <meta charset="UTF-8" />
        <meta http-equiv="X-UA-Compatible" content="IE=edge" />
        <meta name="viewport" content="width=device-width, initial-scale=1.0" />

        <script src="jquery.min.js"></script>
        <style>
            textarea {
            width: 100%;
            height: 100px;
            font-family: Arial, Helvetica, sans-serif;
            font-size: 11px;
            }
        </style>
        </head>
        <body>
        <div><button id="btnLogin">Ajax Call</button></div>
        <textarea id="pId"></textarea>
        <textarea id="resp"></textarea>
        </body>
    
        <script>
        let btn = document.getElementById("btnLogin");
        let p = document.getElementById("pId");
        let resp = document.getElementById("resp");
    
        let token =
            "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJzdWIiOiJqb2tlciIsInNpZCI6IjEiLCJqdGkiOiI3MTVhYjQ4Mi02ZmI3LTQwZDQtYTZkZS00NGM3ZGRiMDE4YmUiLCJleHAiOjE2MjczOTU4NDQsImlzcyI6ImFiaGF5bmF0aCIsImF1ZCI6ImFiaGF5bmF0aCJ9.tLKDoxlIHU6cJYryDckNqerMy1_CnIA4RoLhxxoHxV0";
        const search_path = "http://localhost/redditapi/api/v1/user/search_user";
    
        btn.onclick = function () {
            console.log("--onclick--  final 3");
            let param = { txt: "abhay" };
            p.innerHTML = token;
            $.ajax({
            url: search_path,
            data: JSON.stringify(param),
            type: "POST",
            headers: {
                Authorization: "Bearer " + token,
                "Content-Type": "application/json",
                "Access-Control-Allow-Origin": "*",
            },
            error: function (err) {
                resp.innerHTML = JSON.stringify(err);
            },
            success: function (res) {
                resp.innerHTML = JSON.stringify(res);
            },
            });
        };
        </script>
    </html>

    const arr = ["rohit", "shikhar", "virat"];
    
    console.log(arr.includes("virat")); //true
    console.log(arr.includes("sachin")); //false

Combining arrays


    const players = ["rohit", "shikhar", "virat"];

    const arr = ["hardik", "bhuvneshwar", "ravichandran"];
    
    players.push(...arr);
    
    console.log(JSON.stringify(players)); /* ["rohit","shikhar","virat","hardik","bhuvneshwar","ravichandran"] */

Combining Arrays at specific position


    const players = ["rohit", "shikhar", "virat"];

    const arr = ["hardik", ...players, "bhuvneshwar", "ravichandran"];
    
    console.log(JSON.stringify(arr)); /* ["hardik","rohit","shikhar","virat","bhuvneshwar","ravichandran"] */

Combining Object properties


    const players = {
        captain: "Virat Kohli",
        wicketKeeper: "MS Dhoni"
    };
    
    const india = {
        spinner: "R. Ashvin",
        fast: "Bhuvneshwar Kumar",

    };

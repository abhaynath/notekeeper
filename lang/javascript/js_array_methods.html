Apply function to every element and return new array (map)


    const arr = [1, 2, 3, 4, 5, 6];

    // add one to every element
    const oneAdded = arr.map(num => num + 1);
    console.log(oneAdded); // output [2, 3, 4, 5, 6, 7]
    
    console.log(arr); // output: [1, 2, 3, 4, 5, 6]

Apply function to each element and return one value (reduce)

The reduce() method executes a reducer function (that you provide) on each element of the array, resulting in a single output value. 


    var arr = [1, 2, 3, 4, 5];
    var sum = arr.reduce(function (acc, elem) {
        acc = acc + elem;
        return acc;
    
    });
    console.log("Sum : " + sum); // Sum : 15

Checks if at least one element passes the condition (some)

This method check if at least one of array's item passed the condition. If passed, it return 'true' otherwise 'false'.


    const arr = [1, 2, 3, 4, 5, 6];

    // at least one element is greater than 4?
    const largeNum = arr.some(num => num > 4);
    console.log(largeNum); // output: true
    
    // at least one element is less than or equal to 0?
    const smallNum = arr.some(num => num <= 0);
    console.log(smallNum); // output: false

Check if every element passes the condition (every)

This method check if all array's item passed the condition. If passed, it return 'true' otherwise 'false'.


    const arr = [1, 2, 3, 4, 5, 6];

    // all elements are greater than 4
    const greaterFour = arr.every(num => num > 4);
    console.log(greaterFour); // output: false
    
    // all elements are less than 10
    const lessTen = arr.every(num => num < 10);
    console.log(lessTen); // output: true

Sorting an array (sort)


    const arr = [1, 2, 3, 4, 5, 6];
    const alpha = ['e', 'a', 'c', 'u', 'y'];
    
    // sort in descending order
    descOrder = arr.sort((a, b) => a > b ? -1 : 1);
    console.log(descOrder); // output: [6, 5, 4, 3, 2, 1]
    
    // sort in ascending order
    ascOrder = alpha.sort((a, b) => a > b ? 1 : -1);
    console.log(ascOrder); // output: ['a', 'c', 'e', 'u', 'y']

Combining arrays (spread operator)


    const players = ["rohit", "shikhar", "virat"];

    const arr = ["hardik", "bhuvneshwar", "ravichandran"];
    
    players.push(...arr);
    
    console.log(JSON.stringify(players)); /* ["rohit","shikhar","virat","hardik","bhuvneshwar","ravichandran"] */

Combining Arrays at specific position (spread operator)


    const players = ["rohit", "shikhar", "virat"];

    const arr = ["hardik", ...players, "bhuvneshwar", "ravichandran"];
    
    console.log(JSON.stringify(arr)); /* ["hardik","rohit","shikhar","virat","bhuvneshwar","ravichandran"] */

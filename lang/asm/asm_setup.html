Assembly 

section .data
    msg db "Hello, World!", 0Ah  ; string with newline
    len equ $ - msg              ; length of string

section .text
    global main
    extern printf

main:
    ; Call printf("Hello, World!\n")
    lea rcx, [rel msg]   ; RCX = pointer to msg (first argument in Windows x64 calling convention)
    call printf
    ret

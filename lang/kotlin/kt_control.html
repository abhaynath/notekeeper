Using 'untill'


  var sum = 0
  for (i in 1 until 10) 
  {
      sum += i
  }
  println("Sum = " + sum)

Using increment/ decrement other than 1


  for(i in 0..10 step 3)
  {
    println(i)
  }

When Expression

Example 1


  val run = 200
  val result = when(run){
      100->"century"
      50->"fifty"
      200->"Double Century"
      else->"I have no idea"
  }
  println("result : "+result)

Example 2


  val hour = 2
  val timeOfDay: String

  timeOfDay = when (hour) {
      0, 1, 2, 3, 4, 5 -> "Early morning"
      6, 7, 8, 9, 10, 11 -> "Morning"
      12, 13, 14, 15, 16 -> "Afternoon"
      17, 18, 19 -> "Evening"
      20, 21, 22, 23 -> "Late Evening"
      else -> {
          "Invalid Hour"
      }

  }
  println("Time is = " + timeOfDay)

Example 3 using Ranges


    val hour = 2
    val timeOfDay: String

    timeOfDay = when (hour) {
        in 0..5 -> "Early morning"
        in 6..11 -> "Morning"
        in 12..16 -> "Afternoon"
        in 17..19 -> "Evening"
        in 20..23 -> "Late Evening"
        else -> {
            "Invalid Hour"
        }

    }
    println("Time is = " + timeOfDay)

Multiple Conditions


  val runs = 153
  val result = when{
      runs==0->"Duck"
      runs>=50 && runs<100->"Half Century"
      runs>=100 && runs<200->"Century"
      runs>=200 && runs<300->"Double Century"
      else->"Invalid score"
  }
  println("Score  = "+result)

Function Type Signatures

We can use type aliases to declare function signature.


type GreetFunction = (n: string) => void;

function greeting(name: string, fn: GreetFunction) {
  fn(name);
}

function sayHi(n: string) {
  console.log("Hi, " + n + "!");
}
function sayHello(n: string) {
  console.log("Hello, " + n + "!");
}

greeting("Abhay", sayHi);
greeting("Abhay", sayHello);

Call Signatures


type UserLogin = {
    userName: string;
    password: string;
    onLogin: (name: string, password: string) => void;
    forgotPassword: (emailId: string) => void;
};

let loginScreen: UserLogin = {
    userName: "abhay",
    password: "abhay123",
    onLogin: (n: string, p: string) => {
        console.log(n, p);
    },
    forgotPassword: (email: string) => {
        console.log("email : ", email);
    },
};
loginScreen.onLogin("abc", "abc123");
loginScreen.forgotPassword("abhaynath@gmail.com");

Using multiple generic types


function map<Input, Output>(arr: Input[],fn: (arg: Input) => Output): Output[] {
  return arr.map(fn);
}

const parsed = map([2, 5, 7], (n) => {
  return n * 2;
});

TypeScript

Primitive Types


let myName: string = "Abhay";
let age: number = 32;
let isWorking: boolean = true;
let address: object = {
  city: "Mumbai",
  state: "Maharashtra",
};
let dob = new Date();

console.log(`myName = ${myName}  type = ${typeof myName}`);
console.log(`age = ${age}  type = ${typeof age}`);
console.log(`isWorking = ${isWorking}  type = ${typeof isWorking}`);
console.log(`address = ${JSON.stringify(address)}  type = ${typeof address}`);
console.log(`dob = ${dob}  type = ${typeof dob}`);

Function arguments and return types


function greeting(name: string): string {
    const str = `Hello, ${name} !`;
    return str;
}
console.log(greeting("Abhay"));

Object Types as function arguments and return type


let person = { name: "Abhay", city: "Mumbai" };

function displayInfo(obj: { name: string; city: string }): {
  name: string;
  city: string;
} {
  obj.name = "Mr." + obj.name;
  return obj;
}
let obj = displayInfo(person);
console.log(obj.name);

Optional properties in object


let person = { name: "Abhay" };

function displayInfo(obj: { name: string; city?: string }): {
  name: string;
  city?: string;
} {
  obj.name = "Mr." + obj.name;
  return obj;
}
let obj = displayInfo(person);
console.log(obj.name);

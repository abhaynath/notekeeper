Local State


@Composable
fun ColorBox(modifier: Modifier = Modifier) {
    val color = remember { mutableStateOf(Color.Yellow) }
    Box(modifier = modifier.background(color.value).clickable {
        color.value = Color(
            Random.nextFloat(), Random.nextFloat(), Random.nextFloat(), 1f
        )
    })

}

fun main() = application {
    Window(onCloseRequest = ::exitApplication, title = "Jetpack test") {
        ColorBox(Modifier.fillMaxSize())
    }
}

External State Management


@Composable
fun ColorBox(modifier: Modifier = Modifier, updateColor: (Color) -> Unit) {
    Box(modifier = modifier.background(Color.Red).clickable {
        updateColor(Color(Random.nextFloat(), Random.nextFloat(), Random.nextFloat(), 1f))
    })
}

fun main() = application {
    Window(
        onCloseRequest = ::exitApplication, title = "Postman Desktop"
    ) {
        Column(Modifier.fillMaxSize()) {
            val color = remember { mutableStateOf(Color.Yellow) }
            ColorBox(Modifier.weight(1f).fillMaxSize()) {
                color.value = it;
            }
            Box(
                modifier = Modifier
                    .background(color.value)
                    .weight(1f)
                    .fillMaxSize()
            )
        }
    }
}
